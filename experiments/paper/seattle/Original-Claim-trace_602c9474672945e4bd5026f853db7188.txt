C) Percent change calculation (2025 vs 2024)
SQL executed:
WITH counts AS (
  SELECT
    COUNT(DISTINCT report_number) FILTER (
      WHERE report_datetime >= DATE '2024-06-01'
        AND report_datetime < DATE '2024-09-01'
        AND sector = 'M'
        AND offense_category = 'VIOLENT CRIME'
    ) AS incidents_2024,
    COUNT(DISTINCT report_number) FILTER (
      WHERE report_datetime >= DATE '2025-06-01'
        AND report_datetime < DATE '2025-09-01'
        AND sector = 'M'
        AND offense_category = 'VIOLENT CRIME'
    ) AS incidents_2025
  FROM public.crime_data
)
SELECT
  incidents_2024,
  incidents_2025,
  ROUND(
    CASE WHEN incidents_2024 = 0 THEN NULL
         ELSE ((incidents_2025 - incidents_2024)::numeric / incidents_2024) * 100
    END, 2
  ) AS pct_change_rounded_2dp,
  ROUND(
    CASE WHEN incidents_2024 = 0 THEN NULL
         ELSE ((incidents_2025 - incidents_2024)::numeric / incidents_2024) * 100
    END, 0
  ) AS pct_change_rounded_0dp
FROM counts;

Results:
- incidents_2024 = 105; incidents_2025 = 67
- pct_change_rounded_2dp = -36.19%; pct_change_rounded_0dp = -36%

Percent change (computed earlier)

WITH totals AS (

    EXTRACT(YEAR FROM offense_date)::int AS year,
    COUNT(*) AS violent_offense_rows,

  FROM public.crime_data

SELECT
  (SELECT violent_offense_rows FROM totals WHERE year = 2024) AS offense_rows_2024,
  (SELECT violent_offense_rows FROM totals WHERE year = 2025) AS offense_rows_2025,
  CASE WHEN (SELECT violent_offense_rows FROM totals WHERE year = 2024) IS NULL OR (SELECT violent_offense_rows FROM totals WHERE year = 2024) = 0
