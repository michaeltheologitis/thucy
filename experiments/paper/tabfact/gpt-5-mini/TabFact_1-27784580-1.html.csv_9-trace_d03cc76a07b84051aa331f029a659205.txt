WITH suki AS (
  SELECT REPLACE("ratings", '%', '')::numeric AS suki_rating
  FROM public.tsuki_no_koibito
  WHERE lower("romanized title") = 'suki to ietara iinoni'
),
minr AS (
  SELECT "romanized title" AS min_title, REPLACE("ratings", '%', '')::numeric AS min_rating
  FROM public.tsuki_no_koibito
  ORDER BY REPLACE("ratings", '%', '')::numeric ASC
  LIMIT 1
)
SELECT
  suki.suki_rating,
  minr.min_title,
  minr.min_rating,
  (suki.suki_rating - minr.min_rating) AS diff_points,
  CASE
    WHEN minr.min_rating = 0 THEN NULL
    ELSE ROUND( (suki.suki_rating - minr.min_rating) / minr.min_rating * 100, 6)
  END AS diff_percent
FROM suki CROSS JOIN minr;
