       split_part(regexp_replace(record, '\\s*-\\s*', '-'), '-', 1)::int AS wins,
       split_part(regexp_replace(record, '\\s*-\\s*', '-'), '-', 2)::int AS losses
FROM public.t_1984_85_boston_celtics_season

6) Cross-check: wins and losses across the rows present (by parsing scores)
SQL:
WITH parsed AS (
  SELECT *,
    (regexp_matches(score, '(\d+)\\s*-\\s*(\d+)'))[1]::int AS cscore,
    (regexp_matches(score, '(\d+)\\s*-\\s*(\d+)'))[2]::int AS oscore
  FROM public.t_1984_85_boston_celtics_season
  WHERE score ~ '\\d+\\s*-\\s*\\d+'

SELECT count(*) AS total_games,
  sum(CASE WHEN cscore>oscore THEN 1 ELSE 0 END) AS wins,
  sum(CASE WHEN cscore<oscore THEN 1 ELSE 0 END) AS losses
FROM parsed;
