SELECT venue,
       "home team score" AS home_score_text,
       "away team score" AS away_score_text,
       (trim(both '()' from split_part("home team score",' ',2)))::int AS home_total,
       (trim(both '()' from split_part("away team score",' ',2)))::int AS away_total,
       ((trim(both '()' from split_part("home team score",' ',2)))::int > (trim(both '()' from split_part("away team score",' ',2)))::int) AS home_gt_away
FROM t_1954_vfl_season
ORDER BY date;

2) Count and list of venues with home_total > away_total
Executed SQL:
SELECT COUNT(DISTINCT venue) AS num_venues_home_gt,
       array_agg(DISTINCT venue ORDER BY venue) AS venues
FROM (
   SELECT venue,
          (trim(both '()' from split_part("home team score",' ',2)))::int AS home_total,
          (trim(both '()' from split_part("away team score",' ',2)))::int AS away_total
   FROM t_1954_vfl_season
) s
WHERE home_total > away_total;

Result:
- num_venues_home_gt: 2
- venues: ["junction oval", "kardinia park"]

GROUP BY venue
