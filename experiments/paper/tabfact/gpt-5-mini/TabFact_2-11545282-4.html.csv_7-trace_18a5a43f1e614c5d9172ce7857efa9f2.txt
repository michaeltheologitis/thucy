2) Parse start and end years (expanding 2-digit end years to the same century as the start)

SELECT player, "years for jazz" as raw,
       CAST(trim(split_part("years for jazz", '-',1)) AS integer) as start_year,
       CASE WHEN length(trim(split_part("years for jazz", '-',2))) <= 2
            THEN (CAST(trim(split_part("years for jazz", '-',1)) AS integer)/100)*100 + CAST(trim(split_part("years for jazz", '-',2)) AS integer)
            ELSE CAST(trim(split_part("years for jazz", '-',2)) AS integer)
       END as end_year
FROM public.utah_jazz_all_time_roster
WHERE lower(player) IN ('adrian dantley','brad davis');

- player: adrian dantley, raw: 1979 - 86, start_year: 1979, end_year: 1986
- player: brad davis,    raw: 1979 - 80, start_year: 1979, end_year: 1980

3) Compute how many more years Dantley remained with the Jazz beyond Davis’s end year
SQL executed:
WITH parsed AS (
  SELECT player,
   substring("years for jazz" FROM '([0-9]{4})')::int AS start_year,
   CASE WHEN char_length(substring(trim("years for jazz") FROM '([0-9]{2,4})$')) = 2
      THEN ((substring("years for jazz" FROM '([0-9]{4})')::int/100)*100 + substring(trim("years for jazz") FROM '([0-9]{2,4})$')::int)
      ELSE substring(trim("years for jazz") FROM '([0-9]{2,4})$')::int
