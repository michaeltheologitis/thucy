SELECT name, "dec ( j2000 )" AS declination_text, matches[1]::int AS minutes, matches[2]::int AS seconds

Compute maximum total (MM*60 + SS) and identify maxima (evidence)

WITH parsed AS (
  SELECT
    name,
    "dec ( j2000 )" AS declination_text,
    (regexp_replace("dec ( j2000 )", '.*?([0-9]+)′([0-9]+)″.*', '\1'))::int AS minutes,
    (regexp_replace("dec ( j2000 )", '.*?([0-9]+)′([0-9]+)″.*', '\2'))::int AS seconds,
    ((regexp_replace("dec ( j2000 )", '.*?([0-9]+)′([0-9]+)″.*', '\1'))::int * 60 + (regexp_replace("dec ( j2000 )", '.*?([0-9]+)′([0-9]+)″.*', '\2'))::int) AS total_arcseconds
  FROM dorado_group
  WHERE "dec ( j2000 )" ~ '([0-9]+)′([0-9]+)″'

SELECT name, declination_text, minutes, seconds, total_arcseconds
FROM parsed
WHERE total_arcseconds = (SELECT MAX(total_arcseconds) FROM parsed);
