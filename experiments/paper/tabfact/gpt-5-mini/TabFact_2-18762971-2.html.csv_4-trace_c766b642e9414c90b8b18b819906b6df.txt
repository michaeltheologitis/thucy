Step 2 — Compute day gaps between consecutive matches and find the minimum
Executed SQL (produce per-match gaps):
WITH parsed AS (
  SELECT date, opponent, to_date(date, 'FMDD FMMonth YYYY') AS parsed_date
  FROM public.t_2000_01_sheffield_wednesday_f_c_season
)
SELECT date AS date_text,
       parsed_date,
       opponent,
       lead(parsed_date) OVER (ORDER BY parsed_date) AS next_date,
       lead(opponent) OVER (ORDER BY parsed_date) AS next_opponent,
       (lead(parsed_date) OVER (ORDER BY parsed_date) - parsed_date) AS gap_days
FROM parsed
ORDER BY parsed_date;

  FROM public.t_2000_01_sheffield_wednesday_f_c_season
