2) Compute maximum consecutive losses in November
SQL:
WITH nov AS (
  SELECT game, score,
    row_number() OVER (ORDER BY game) AS rn,
    row_number() OVER (PARTITION BY (score LIKE 'l%') ORDER BY game) AS rn_flag,
    (row_number() OVER (ORDER BY game) - row_number() OVER (PARTITION BY (score LIKE 'l%') ORDER BY game)) AS grp,
    (score LIKE 'l%') AS is_loss
  FROM public.t_2009_10_denver_nuggets_season
  WHERE lower(date) LIKE 'november%'
  ORDER BY game
)
SELECT COALESCE(MAX(cnt), 0) AS max_consecutive_losses FROM (
  SELECT grp, COUNT(*) AS cnt
  FROM nov
  WHERE is_loss
  GROUP BY grp
) s;

Result:
- max_consecutive_losses = 1
