SELECT CASE
  WHEN upper(trim(split_part("city of license", ',', 2))) IN ('IL','ILLINOIS') THEN 'Illinois'

END AS state_norm,
count(*) AS cnt
FROM whlp
GROUP BY state_norm
ORDER BY cnt DESC;
