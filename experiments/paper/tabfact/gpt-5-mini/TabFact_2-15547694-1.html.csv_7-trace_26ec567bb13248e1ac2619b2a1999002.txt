agg AS (
 SELECT origin,
  SUM(cnt) AS total_members,
  COUNT(*) FILTER (WHERE cnt>0) AS num_parties_represented
 FROM unpivot
 GROUP BY origin
),
ranked AS (
 SELECT origin, party, cnt,
  ROW_NUMBER() OVER (PARTITION BY origin ORDER BY cnt DESC, party) AS rn

)
SELECT a.origin,

 a.num_parties_represented,

LEFT JOIN ranked r1 ON a.origin=r1.origin AND r1.rn=1
LEFT JOIN ranked r2 ON a.origin=r2.origin AND r2.rn=2
ORDER BY a.origin;
