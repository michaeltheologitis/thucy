WITH ar AS (
  SELECT replace("ratings",'%','')::numeric AS ar_rating

), earliest AS (

  FROM tsuki_no_koibito

SELECT ar.ar_rating AS arienai_kisu_rating,
       earliest.e_rating AS earliest_rating,
