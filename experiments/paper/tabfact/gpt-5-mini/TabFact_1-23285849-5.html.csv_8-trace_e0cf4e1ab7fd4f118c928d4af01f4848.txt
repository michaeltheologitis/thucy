), flagged AS (
  SELECT *,
    LAG(is_loss) OVER (ORDER BY game) AS prev_is_loss

  SELECT *,
    CASE WHEN is_loss = 1 AND COALESCE(prev_is_loss, 0) = 0 THEN 1 ELSE 0 END AS streak_start
  FROM flagged
), grouped AS (
  SELECT *,
    SUM(streak_start) OVER (ORDER BY game ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS streak_id
