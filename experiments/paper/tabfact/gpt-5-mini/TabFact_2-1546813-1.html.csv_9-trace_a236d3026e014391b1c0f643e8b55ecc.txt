2) Compute min and max winning-team scores (parse left side; treat ½ as 0.5)
SQL:
SELECT
  MIN(left_score) AS min_score,
  MAX(left_score) AS max_score
FROM (
  SELECT
    CAST(
      TRIM(REPLACE(SPLIT_PART(score, '-', 1), '½', '.5'))
    AS NUMERIC) AS left_score
  FROM public.solheim_cup
  WHERE score IS NOT NULL
) t;
Result:
- min_score = 11.5
- max_score = 18

3) Verify the left side is always the winning-team score (left > right)
SQL:
SELECT year, "winning team", score,
       CAST(REPLACE(TRIM(SPLIT_PART(score, ' - ', 1)), '½', '.5') AS numeric) AS parsed_left_score,
       CAST(REPLACE(TRIM(SPLIT_PART(score, ' - ', 2)), '½', '.5') AS numeric) AS parsed_right_score,
       CASE WHEN CAST(REPLACE(TRIM(SPLIT_PART(score, ' - ', 1)), '½', '.5') AS numeric) > CAST(REPLACE(TRIM(SPLIT_PART(score, ' - ', 2)), '½', '.5') AS numeric) THEN true ELSE false END AS left_greater
FROM public.solheim_cup
ORDER BY year;
Result (all rows): left_greater = true

4) List rows achieving the min and max winning-team scores
SQL:
WITH parsed AS (
  SELECT year,
         "winning team",
         score,
         CAST(REPLACE(TRIM(SPLIT_PART(score, ' - ', 1)), '½', '.5') AS numeric) AS parsed_left_score
  FROM public.solheim_cup
)
SELECT * FROM parsed
WHERE parsed_left_score = (SELECT MIN(parsed_left_score) FROM parsed)
