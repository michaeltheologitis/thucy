SELECT DISTINCT team

         NULLIF(regexp_replace(trim(split_part("home team score",'.',1)), '[^0-9]', '', 'g'), '')::int AS goals
  FROM public.t_1938_vfl_season
  UNION ALL

         NULLIF(regexp_replace(trim(split_part("away team score",'.',1)), '[^0-9]', '', 'g'), '')::int AS goals
  FROM public.t_1938_vfl_season
) t
WHERE goals < 10
ORDER BY team;
