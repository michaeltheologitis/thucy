4) Compute point difference and relative percent difference
SQL:
WITH f AS (
  SELECT "romanized title" AS fuku_title,
         to_date("broadcast date", 'FMMonth DD , YYYY') AS fuku_date,
         regexp_replace(ratings, '%', '', 'g')::numeric AS fuku_rating
  FROM public.tsuki_no_koibito
  WHERE "romanized title" = 'fukusyū no puropōzu'
), l AS (
  SELECT "romanized title" AS latest_title,
         to_date("broadcast date", 'FMMonth DD , YYYY') AS latest_date,
         regexp_replace(ratings, '%', '', 'g')::numeric AS latest_rating
  FROM public.tsuki_no_koibito
  ORDER BY to_date("broadcast date", 'FMMonth DD , YYYY') DESC
  LIMIT 1
)
SELECT f.fuku_title,
       f.fuku_date,
       f.fuku_rating,
       l.latest_title,
       l.latest_date,
       l.latest_rating,
       (f.fuku_rating - l.latest_rating) AS point_diff,
       ((f.fuku_rating - l.latest_rating) / l.latest_rating * 100) AS percent_diff
FROM f CROSS JOIN l;
