Verification analysis
Step 1: Parse text dates, compute each sport’s earliest date, and rank ascending.
Executed SQL (exact):
WITH normalized AS (
  SELECT sport,
         date,
         to_date(initcap(lower(regexp_replace(date, '\s*,\s*', ',', 'g'))), 'FMMonth FMDD, YYYY') AS parsed_date
  FROM public.iowa_corn_cy_hawk_series
),
port_earliest AS (
  SELECT sport, min(parsed_date) AS earliest_date
  FROM normalized
  GROUP BY sport
)
SELECT row_number() OVER (ORDER BY earliest_date, sport) AS rank,
       sport,
       earliest_date
FROM sport_earliest
ORDER BY earliest_date, sport;
