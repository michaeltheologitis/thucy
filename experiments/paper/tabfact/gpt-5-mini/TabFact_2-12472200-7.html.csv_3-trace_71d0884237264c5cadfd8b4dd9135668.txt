SELECT year,
       rank,
       LAG(rank) OVER (ORDER BY year::int) AS prev_rank,
       (rank - LAG(rank) OVER (ORDER BY year::int)) AS change,

         WHEN rank < LAG(rank) OVER (ORDER BY year::int) THEN TRUE
         ELSE FALSE
       END AS improved
FROM ranks
ORDER BY year::int;
