3) Parse dates and compare whether "aaliyah" aired later
SQL:
WITH a AS (
  SELECT "original air date" AS a_text,
         to_date(initcap(regexp_replace("original air date",'\\s*,\\s*',',')), 'FMMonth DD, YYYY') AS a_date,
         "nick prod" AS a_nick
  FROM list_of_all_that_episodes
  WHERE lower("episode title") = 'aaliyah'
), b AS (
  SELECT "original air date" AS b_text,
         to_date(initcap(regexp_replace("original air date",'\\s*,\\s*',',')), 'FMMonth DD, YYYY') AS b_date,
         "nick prod" AS b_nick
  FROM list_of_all_that_episodes
  WHERE "nick prod" = (SELECT a_nick + 1 FROM a)
)
