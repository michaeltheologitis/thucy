SELECT initcap(month) AS month, count(*) AS count
FROM (
  SELECT substring(lower("original airdate") FROM '^\s*([a-z]+)') AS month
  FROM public.list_of_republic_of_doyle_episodes
) AS t
WHERE month IS NOT NULL
GROUP BY month
ORDER BY count DESC;

- March: 5
