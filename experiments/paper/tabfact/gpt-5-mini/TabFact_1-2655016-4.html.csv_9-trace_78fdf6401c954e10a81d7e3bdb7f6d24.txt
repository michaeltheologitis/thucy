3) Premiere/finale (min/max) parsed dates per season
SQL:
WITH parsed AS (
 SELECT season,
  to_date(regexp_replace(initcap(trim("original air date")), '\s+,', ','), 'Month DD, YYYY') AS parsed_date
 FROM list_of_all_that_episodes
 WHERE season IN (2,10)
)
SELECT season,
 MIN(parsed_date) AS premiere_date,
 to_char(MIN(parsed_date),'FMMonth') AS premiere_month,
 MAX(parsed_date) AS finale_date,
 to_char(MAX(parsed_date),'FMMonth') AS finale_month
FROM parsed
GROUP BY season
ORDER BY season;
