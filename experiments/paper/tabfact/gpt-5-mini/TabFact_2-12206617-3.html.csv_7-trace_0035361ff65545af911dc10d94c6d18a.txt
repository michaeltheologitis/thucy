2) Compute W/L for the final three May games by comparing cumulative wins between consecutive games
SQL executed:
WITH may AS (
  SELECT date, (regexp_replace(lower(date),'^[a-z]+\s+([0-9]{1,2}).*$','\1'))::int AS day, opponent, score, record,
         (trim(split_part(record,'-',1)))::int AS wins
  FROM public.t_1991_toronto_blue_jays_season
  WHERE lower(date) LIKE 'may %'
), computed AS (
  SELECT date, opponent, score, record, day,
         CASE WHEN lag(wins) OVER (ORDER BY day) IS NULL THEN NULL
              WHEN wins > lag(wins) OVER (ORDER BY day) THEN 'W' ELSE 'L' END AS result
  FROM may
)
SELECT date, opponent, score, record, result
FROM computed
WHERE day >= (SELECT max(day) FROM may) - 2
ORDER BY day;
