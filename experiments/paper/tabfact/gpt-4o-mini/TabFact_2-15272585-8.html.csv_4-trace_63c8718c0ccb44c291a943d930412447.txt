Step 3: Compute games margin per loss
Method: Parse each score string set-by-set, summing (opponent games − Voskoboeva games) across sets in that match.
SQL executed:
WITH matches AS (
  SELECT date, opponent, score, replace(score, ' ', '') AS score_clean
  FROM public.galina_voskoboeva
  WHERE lower(outcome) = 'runner - up'
),
sets AS (
  SELECT date, opponent, score, unnest(string_to_array(score_clean, ',')) AS set_str
  FROM matches
),
parsed AS (
  SELECT date, opponent, score,
         CAST(split_part(set_str, '-', 1) AS int) AS my_games,
         CAST(split_part(set_str, '-', 2) AS int) AS opp_games
  FROM sets
)
SELECT date, opponent, score,
       SUM(opp_games - my_games) AS games_margin
FROM parsed
GROUP BY date, opponent, score
ORDER BY date ASC;
