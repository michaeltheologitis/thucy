3) Compute the absolute minimum attendance and show the matching game(s)
SQL:
WITH ParsedAttendance AS (
    SELECT
        date,
        team,
        TRIM(SUBSTRING("location attendance" FROM '^(.*?)(\s\d+)$')) AS venue,
        CAST(TRIM(SUBSTRING("location attendance" FROM '\s(\d+)$')) AS INTEGER) AS attendance,
        game
    FROM public.t_2010_11_detroit_pistons_season)
,
MinAttendance AS (
    SELECT MIN(attendance) AS min_attendance
    FROM ParsedAttendance
)
SELECT PA.date,
       PA.team,
       PA.venue,
       PA.attendance,
       PA.game
FROM ParsedAttendance PA
JOIN MinAttendance MA ON PA.attendance = MA.min_attendance;
