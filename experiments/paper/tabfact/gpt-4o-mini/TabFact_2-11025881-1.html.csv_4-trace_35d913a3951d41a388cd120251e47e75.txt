Executed SQL (primary evidence)
-- Single query computing the gap between first and last by rank
WITH ranked_competitors AS (
    SELECT 
        rank,
        name,
        points
    FROM 
        t_1976_world_junior_figure_skating_championships
),
first AS (
    SELECT 
        name AS first_name, 
        points AS first_points
    FROM 
        ranked_competitors
    WHERE 
        rank = (SELECT MIN(rank) FROM ranked_competitors)
),
last AS (
    SELECT 
        name AS last_name, 
        points AS last_points
    FROM 
        ranked_competitors
    WHERE 
        rank = (SELECT MAX(rank) FROM ranked_competitors)
)
SELECT 
    first.first_name,
    first.first_points,
    last.last_name,
    last.last_points,
    (first.first_points - last.last_points) AS gap,
    ABS(first.first_points - last.last_points) AS absolute_gap 
FROM 
    first, last;
