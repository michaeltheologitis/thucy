WITH may AS (
  SELECT
    date,
    opponent,
    score,
    loss,
    record,
    CAST(split_part(record, ' - ', 1) AS int) AS wins,
    CAST(split_part(record, ' - ', 2) AS int) AS losses,
    CAST(regexp_replace(date, '[^0-9]', '', 'g') AS int) AS day
  FROM public.t_1991_toronto_blue_jays_season
  WHERE lower(date) LIKE 'may %'
),
ordered AS (
  SELECT *,
         LAG(wins) OVER (ORDER BY day) AS prev_wins,
         LAG(losses) OVER (ORDER BY day) AS prev_losses
  FROM may
),
classified AS (
  SELECT
    date,
    opponent,
    score,
    record,
    day,
    CASE
      WHEN prev_wins IS NULL THEN NULL
      WHEN wins = prev_wins + 1 THEN 'W'
      WHEN losses = prev_losses + 1 THEN 'L'
      ELSE NULL
    END AS result
  FROM ordered
)
SELECT date, opponent, score, record, result
FROM classified
ORDER BY day DESC
