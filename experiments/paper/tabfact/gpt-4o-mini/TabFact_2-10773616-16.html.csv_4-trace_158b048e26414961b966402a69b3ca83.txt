2) Extracting numeric points and comparing home vs away for that date
SQL:
SELECT venue, 
       "home team", 
       "away team", 
       CAST(SUBSTRING("home team score" FROM '\((\d+)\)') AS INT) AS home_points, 
       CAST(SUBSTRING("away team score" FROM '\((\d+)\)') AS INT) AS away_points, 
       (CAST(SUBSTRING("home team score" FROM '\((\d+)\)') AS INT) > CAST(SUBSTRING("away team score" FROM '\((\d+)\)') AS INT)) AS home_points_greater 
FROM public.t_1954_vfl_season
