3) Direct computation of rating difference (absolute and relative)
SQL:
WITH f AS (
  SELECT CAST(replace(ratings, '%', '') AS numeric) AS rating
  FROM tsuki_no_koibito
  WHERE "romanized title" = 'fukusyū no puropōzu'
), m AS (
  SELECT CAST(replace(ratings, '%', '') AS numeric) AS rating
  FROM tsuki_no_koibito
  ORDER BY 
    CAST(regexp_replace(split_part("broadcast date", ',', 2), '\\D', '', 'g') AS INT) DESC,
    CASE lower(split_part("broadcast date", ' ', 1))
        WHEN 'january' THEN 1 WHEN 'february' THEN 2 WHEN 'march' THEN 3 WHEN 'april' THEN 4 WHEN 'may' THEN 5 WHEN 'june' THEN 6 WHEN 'july' THEN 7 WHEN 'august' THEN 8 WHEN 'september' THEN 9 WHEN 'october' THEN 10 WHEN 'november' THEN 11 WHEN 'december' THEN 12 END DESC,
    CAST(regexp_replace(split_part(split_part("broadcast date", ',', 1), ' ', 2), '\\D', '', 'g') AS INT) DESC
  LIMIT 1
)
SELECT f.rating AS fuku_rating, m.rating AS most_recent_rating,
       ROUND(f.rating - m.rating, 1) AS point_diff,
       ROUND(((f.rating - m.rating) / m.rating) * 100, 2) AS pct_diff
FROM f, m;
Result: fuku_rating = 15.6, most_recent_rating = 13.4, point_diff = 2.2, pct_diff = 16.42%
