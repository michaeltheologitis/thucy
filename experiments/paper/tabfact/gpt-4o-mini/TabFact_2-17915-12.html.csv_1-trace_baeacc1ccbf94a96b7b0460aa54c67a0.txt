WITH split_sets AS (

cleaned_sets AS (

        regexp_replace(regexp_replace(set_str, '\(.*\)', '', 'g'), '\s+', '', 'g') AS clean_set

),
parsed_scores AS (
    SELECT 
        clean_set,
        split_part(clean_set, '-', 1)::int AS left_score,
        split_part(clean_set, '-', 2)::int AS right_score

)
SELECT 
    left_score, 
    right_score, 
    COUNT(*) AS qualifying_sets,
